\section{Farm Vertical Disintegration and Productivity}
\roadmapslide

% --------------------------------------------
\subsection{Motivation}

\begin{frame}{Paper Overview}

\textbf{Research question:} Does vertical disintegration---the emergence of industries supplying production services to farms---contribute to agricultural productivity?

\bigskip

\textbf{Why this matters:}
\begin{itemize}
    \item Agricultural TFP growth is exceptional (1.49\% annually, 1948--2021)
    \item Typically attributed to R\&D and technology adoption
    \item Organizational change as a source of productivity growth is understudied
\end{itemize}

\bigskip

\textbf{What we do:}
\begin{itemize}
    \item First systematic estimates of services--yield relationship
    \item Leverage restricted Census Bureau administrative data
    \item Document heterogeneity by crop type and farm scale
    \item Examine longitudinal dynamics
\end{itemize}

\end{frame}
